import re
from nltk.corpus import stopwords
from nltk.stem import LancasterStemmer
from multiprocessing import Pool
 
def data_text_cleaning(data):
    # html 태그 제거
    only_string = re.sub('[/\u2009/\n]','', data)
    # 영문자 이외 문자는 공백으로 변환
    only_english = re.sub('[^a-zA-Z]', ' ', only_string)
    
    # 소문자 변환
    no_capitals = only_english.lower().split(' ')
    
    # 불용어 제거
    stops = set(stopwords.words('english'))
    no_stops = [word for word in no_capitals if not word in stops if len(word)>1]
    
    stemmer = nltk.stem.SnowballStemmer('english')
    stemmer_words = [stemmer.stem(word) for word in no_stops]
    
    return ' '.join(stemmer_words)
 
def use_multiprocess(func, iter, workers):
    pool = Pool(processes=workers)
    result = pool.map(func, iter)
    pool.close()
    return result

사용자 증상 데이터 전처리 중 시행착오


(1) 키워드 선정
1) 각 전문과목별로 라벨링되어 있는 데이터셋의 부재 -> 직접 keyword를 각 과목의 학회 및 Q&A를 읽어보고 선정
2) 이후 선정된 키워드로 크롤링을 진행 : 네이버 지식인, 네이버 포스트, 네이버 카페
3) 그러나 각 키워드로 모은 내용 중 광고가 포함된 경우도 많고 내용들이 생각보다 매우 길어 LSTM으로 학습하기에 어려움이 있었다.

(2) 데이터 재수집
1) 내용이 생각보다 매우 길었고, 광고 등 상관없으면서 쓸모없는 데이터도 많았다.
2) 따라서 먼저 '제목'만 가지고 학습을 진행하였고, 모델의 성능이 어느 정도 작동함을 확인하였다. (정확도 약 64%)
3) 데이터 라벨링을 보다 정확히 하여 성능을 높이도록 하기 위하여 네이버 지식인 데이터에서 '제목'이 일치하는 경우를 골라 '질문' 내용만 수집하였다.
4) 많은 다른 크롤링 후보군의 사이트들을 (다음, 하이닥, 네이버 포스트, 네이버 카페 등) 고려하였으나, 전부 제목 일치가 되지 않거나 불필요한 정보가 너무 많이 나오는 등의 문제로 네이버 지식인과 다음 팁만 재수집하였다.
5) 데이터 재수집 후 카카오의 Khaii를 이용하였고, 그 결과 정확도를 약 71%로 올릴 수 있었다.

(3) 형태소 분석
1) Khaii, Komoran, Okt, Mecab 등 다양한 형태소 분석을 data에 적용하였고, 가장 성능이 잘 나오면서 근거를 잘 보여주는 형태소 분석기를 사용하였다.
2) Khaii는 맥과 리눅스 환경에서만 돌아가는 문제, KKma는 속도 문제 때문에 실질적으로 사용할 수 없었다.
3) 결과적으로 Komoran을 사용하였고 정확도를 77%까지 올릴 수 있었다.
